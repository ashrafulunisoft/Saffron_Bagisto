```php

        'system_name',
        'type',
        'api_key',

    ];

     * Get the status of the support system.
```

```php

{
```

```php
    protected $fillable = [
        'feedback_type',
        'feedback',
        'submitted_by',

    ];

     * Get the status of the feedback.
     */
    public function getStatusTextAttribute()
    {
        return ucfirst($this->status);
    }

     * Scope a query to only include 'new' feedback.
```
