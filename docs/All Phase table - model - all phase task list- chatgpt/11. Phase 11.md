### 2. **Server Configuration Model**

```php
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;

class ServerConfiguration extends Model
{
    use HasFactory;

    protected $fillable = [
        'server_name',
        'ip_address',
        'role',
        'cpu_cores',
        'ram_gb',
        'storage_gb',
        'os',
        'is_active',
    ];

    /**
     * Get the server status (active/inactive).
     */
    public function getStatusAttribute()
    {
        return $this->is_active ? 'Active' : 'Inactive';
    }

    /**
     * Get the server specifications formatted.
     */
    public function getSpecificationsAttribute()
    {
        return "{$this->cpu_cores} cores, {$this->ram_gb} GB RAM, {$this->storage_gb} GB storage";
    }

    /**
     * Scope a query to only include active servers.
     */
    public function scopeActive($query)
    {
        return $query->where('is_active', true);
    }
}
```

```php
     * Scope a query to only include cloud backups.
     */
    public function scopeCloudBackups($query)
    {
        return $query->where('storage_type', 'cloud');

     * Scope a query to only include backups with a specific retention period.
```

### 8. **Monitoring and Logging Configuration Model**

```php
<?php

namespace App\Models;

use Illuminate\Database\Eloquent\Factories\HasFactory;
use Illuminate\Database\Eloquent\Model;

class MonitoringAndLoggingConfiguration extends Model
{
    use HasFactory;

    protected $fillable = [
        'monitoring_tool',
        'logging_tool',
        'alerting_tool',
    ];

    /**
     * Get the monitoring tool.
     */
    public function getMonitoringToolTextAttribute()
    {
        return ucfirst($this->monitoring_tool);
    }

    /**
     * Get the logging tool.
     */
    public function getLoggingToolTextAttribute()
    {
        return ucfirst($this->logging_tool);
    }

    /**
     * Scope a query to include configurations with specific tools.
     */
    public function scopeUsingTool($query, $tool)
    {
        return $query->where('monitoring_tool', $tool)
                     ->orWhere('logging_tool', $tool)
                     ->orWhere('alerting_tool', $tool);
    }
}
```
